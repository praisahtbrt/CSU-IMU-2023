Factors (2) 

# add factor for blood group
blood <- factor(c("O", "AB", "A"),
                levels = c("A", "B", "AB", "O"))
blood
## [1] O  AB A 
## Levels: A B AB O
# add ordered factor for severity
symptoms <- factor(c("SEVERE", "LIGHT", "MODERATE"),
                   levels = c("LIGHT", "MODERATE", "SEVERE"),
                   ordered = TRUE)
symptoms
## [1] SEVERE   LIGHT    MODERATE
## Levels: LIGHT < MODERATE < SEVERE
